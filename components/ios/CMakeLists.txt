vkb__register_component(
        NAME ios_platform
        HEADERS
        include/ios/context.hpp
        SRC
        src/context.mm
        src/entrypoint.cpp
        LINK_LIBS
        vkb__core
)

        NO_CATCH2
        COMPONENT ios_platform
        NAME ios_platform
        SRC
        tests/entrypoint.test.cpp
        LINK_LIBS
        vkb__ios_platform
)
